# GIT COMMANDS

# GIT Conflict Troubleshooting
1. git checkout "branch"
2. git pull
3. git rebase develop
4. code . (resolve conflicts)
5. git add .
6. git rebase --continue
7. git commit --amend or git commit -m ""
8. git push or git push -f
9. git merge develop
